# Installation

# Backend (Laravel)

# Install dependencies
composer install

# Copy and configure environment file
cp .env.example .env
# Edit .env to set your database credentials

# Run migrations
php artisan migrate

# Start the Laravel server
php artisan serve

# Frontend (React)

# Install dependencies
npm install

# Start the development server
npm run dev
